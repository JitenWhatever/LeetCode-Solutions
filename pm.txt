struct page{
        int num;
        int time;
        };

int findnum(vector<struct page> &V, int val){
        int size = V.size();
        for(int i = 0; i < size; i++){
                if(V[i].num == val)
                   return i + 1;

        }
        return false;
        }

int maxtime(vector<struct page> &V){
        int size = V.size();
        int min = 0;
        for(int i = 0; i < size; i++){
                if(V[min].time > V[i].time)
                        min = i;
        }
        return min;
        }
